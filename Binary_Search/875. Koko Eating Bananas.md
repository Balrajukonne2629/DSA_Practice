The approach is to use the binary search to find the minimum value

2. Here to find the total hours needed to eat the bananas can be calculated efficiently by using this one i.e. **(pile+k-1)/k**for every value of the array we should calculate this and sum of their results to get the total hours.
3.  here the variable k is the mid value and iterate the while loop(low<high) conditon.
4. the mid value is calculated by **(low+(high-low)/2)** 
5. and there is a condition function *isValid* which is used to check whether the total hours calculated are valid by comparing with the given hours.
6. if valid then to find the minimum value of k we will **assign the high value as *high=mid***
7. if not then **low=mid+1**
8. finally return the low value.

class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int h) {
        int maxPile = 0;
        
        int k = 0;
        for (int i = 0; i < piles.size(); i++) {
            maxPile = max(maxPile, piles[i]);
        }
        int low = 1, high = maxPile;
        while (low < high) {
            int mid = low + (high - low) / 2;
            int total_hours = pileSum(piles, mid);
            if (total_hours <= h) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        return low;
    }
    int pileSum(vector<int>& piles, int mid) {
        int pileSum = 0;
        for (int i = 0; i < piles.size(); i++) {

            pileSum += (piles[i] + mid - 1) / mid;
        }
        return pileSum;
    }
};
